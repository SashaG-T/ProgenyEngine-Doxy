To Freeze and Thaw the controller you must use the engine registered user action: FreezeEvent

GraphicsInterface& graphicsInterface(Engine::getGraphicsInterface());
uint32_t freezeEvent(graphicsInterface.registerAction("FreezeEvent");
//Freeze every FreezableController.
graphicsInterface.pushEvent(freezeEvent, FreezableController<YourController>::UserEventCode::FREEZE, 0, 0);
//Thaw every FreezableController.
graphicsInterface.pushEvent(freezeEvent, FreezableController<YourController>::UserEventCode::THAW, 0, 0);
